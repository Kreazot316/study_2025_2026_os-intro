\begin{frame}[fragile]{Теоретическое введение}
\phantomsection\label{ux442ux435ux43eux440ux435ux442ux438ux447ux435ux441ux43aux43eux435-ux432ux432ux435ux434ux435ux43dux438ux435}
\begin{itemize}[<+->]
\tightlist
\item
  \textbf{GitHub Pages} --- сервис для размещения статических
  веб-страниц из репозиториев GitHub.
\item
  \textbf{Hugo} --- генератор статических сайтов, превращающий исходные
  файлы в готовый HTML, CSS и JavaScript.
\item
  \textbf{GitHub Actions} --- автоматизация сборки и публикации: при
  обновлении ветки \texttt{main} запускается процесс, который собирает
  сайт и размещает его на Pages.
\item
  \textbf{Wowchemy (Hugo Academic Theme)} --- стартовый шаблон на основе
  Hugo. Для обработки стилей через TailwindCSS необходимы Node.js и
  pnpm.
\end{itemize}
\end{frame}
